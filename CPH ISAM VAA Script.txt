--PULL MOST RECENT MONTH'S ACCOUNTS BY REP
CREATE OR REPLACE TEMPORARY TABLE TOTAL_GP AS 
SELECT  DISTINCT PAYEE_ID,
        PAYEE_NAME,
        ACCOUNT_ID,
        COUNT(*) AS COUNT
FROM    SBX_PSAS_DB.SALES_OPS_GOV.FY25_LEADERBOARD_ISAM_CPH
WHERE PERIOD='2025-10 JAN' --MOST RECENT MONTH'S ACCOUNT
AND PRODUCT_SEGMENT <>'GOALS'
GROUP BY PAYEE_ID,
        PAYEE_NAME,
        ACCOUNT_ID;
--SELECT * FROM TOTAL_GP;

CREATE OR REPLACE TEMPORARY TABLE TOTAL_ACCT AS
SELECT DISTINCT PAYEE_ID AS PAYEEID,
PAYEE_NAME AS PAYEENAME,
COUNT(ACCOUNT_ID) AS TOTAL_ACCT
FROM TOTAL_GP
GROUP BY 1,2;
--SELECT * FROM TOTAL_ACCT;

CREATE OR REPLACE TEMPORARY TABLE HM AS
SELECT CUST_ACCT_ID*1 AS CUST_ACCT_ID_NEW,
SUM(CASE WHEN YR_MTH = '2/1/2025' THEN HM ELSE 0 END) AS HM_COUNT_CURRENT,
SUM(CASE WHEN YR_MTH = '1/1/2025' THEN HM ELSE 0 END) AS HM_COUNT_PRIOR
FROM SBX_PSAS_DB.ANALYTICS.T_CPH_VAA_HM_TEST
WHERE COMP_FLG='Y'
AND YR_MTH IN ('2025-02-01','2025-01-01') 
GROUP BY CUST_ACCT_ID;
--SELECT * FROM HM;

CREATE OR REPLACE TEMPORARY TABLE HMA AS
SELECT CUST_ACCT_ID*1 AS CUST_ACCT_ID_NEW,
SUM(CASE WHEN YR_MTH = '2/1/2025' THEN HMA ELSE 0 END) AS HMA_COUNT_CURRENT,
SUM(CASE WHEN YR_MTH = '1/1/2025' THEN HMA ELSE 0 END) AS HMA_COUNT_PRIOR
FROM (SELECT DISTINCT * FROM SBX_PSAS_DB.ANALYTICS.T_CPH_VAA_HMA)
WHERE COMP_FLG='Y'
AND YR_MTH IN ('2025-02-01','2025-01-01') 
GROUP BY CUST_ACCT_ID;
--SELECT * FROM HMA;

--JOIN GOALS,ACTUALS,TOTAL ACCT WITH HIERARCHY
CREATE OR REPLACE TEMPORARY TABLE COMB AS
SELECT A.CUST_ACCT_ID,
A.SLS_TERR_ID,
A.HOME_DC_ID,
A.REP_NUMBER,
A.REP_NAME,
A.REP_TITLE,
A.CUST_NAME,
A.CUST_CTY,
A.CUST_ST,
A.CUST_ZIP,
A.VPS_ID,
A.VPS_NAME,
A.AVP_ID,
A.AVP_NAME,
A.YR_MTH,
SUM(B.HM_GOAL) AS HM_GOAL,
SUM(B.HMA_GOAL) AS HMA_GOAL,
SUM(C.TOTAL_ACCT) AS TOTAL_ACCT,
SUM(D.HM_COUNT_CURRENT) AS HM_COUNT_CURRENT,
SUM(E.HMA_COUNT_CURRENT) AS HMA_COUNT_CURRENT,
SUM(D.HM_COUNT_PRIOR) AS HM_COUNT_PRIOR,
SUM(E.HMA_COUNT_PRIOR) AS HMA_COUNT_PRIOR
FROM HIERARCHY A 
LEFT JOIN GOALS B
ON A.VPS_ID=B.PAYEE_ID
LEFT JOIN TOTAL_ACCT C
ON A.REP_NUMBER=C.PAYEEID
LEFT JOIN HM D
ON A.CUST_ACCT_ID=D.CUST_ACCT_ID_NEW
LEFT JOIN HMA E
ON A.CUST_ACCT_ID=E.CUST_ACCT_ID_NEW
GROUP BY A.CUST_ACCT_ID,
A.SLS_TERR_ID,
A.HOME_DC_ID,
A.REP_NUMBER,
A.REP_NAME,
A.REP_TITLE,
A.CUST_NAME,
A.CUST_CTY,
A.CUST_ST,
A.CUST_ZIP,
A.VPS_ID,
A.VPS_NAME,
A.AVP_ID,
A.AVP_NAME,
A.YR_MTH;
--SELECT * FROM COMB;

--FINAL TABLE
CREATE OR REPLACE TABLE SBX_PSAS_DB.SALES_OPS_GOV.FY24_VAA_ISAM AS
SELECT * FROM COMB;
